  `dormitorio` INT NULL,
  `lavado` INT NULL,
  `jardin` INT NULL,
  `patio` INT NULL,
  `terraza` INT NULL,
  `alberca` INT NULL,
  `estacionamiento` INT NULL,
  `tipo_vivienda` VARCHAR(20) NULL,
  `techo_vivienda` VARCHAR(20) NULL,
  `piso_vivienda` VARCHAR(20) NULL,
  `serv_luz` TINYINT NULL,
